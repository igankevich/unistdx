118  end_packet() {
119  const std::streamsize ps = this->opacketsize();
120  const std::streamsize hs = this->_oheadersize;
121  // check if the header size has changed
122  // depending on the length of the payload
123  const std::streamsize new_hs = this->overwrite_header(ps);
124  assert(new_hs <= hs);
125  if (new_hs < hs) {
126  const std::streamsize delta = new_hs - hs;
127  char_type* src = this->opacket_begin();
128  // move payload to match new header size
129  traits_type::move(src + new_hs, src + hs, ps - hs);
130  // advance put pointer
131  this->pbump(delta);
132  }
133  #if !defined(NDEBUG) && defined(UNISTDX_DEBUG_PACKETBUF)
134  sys::debug_message(

137  hs,
138  ps-hs

157  read_packet() {
158  state_type old_state;
159  do {
160  old_state = this->_state;
161  switch (this->_state) {
162  case state_type::initial: this->getheader(); break;
163  case state_type::header_is_ready: this->getpayload(); break;
164  case state_type::payload_is_ready:

166  this->finish();

171  }

173  return this->_state == state_type::payload_is_ready;

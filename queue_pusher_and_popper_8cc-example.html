queue_pusher_and_popper.cc

Push all command line arguments to a queue, pop each element and print it to stdout.

#include <algorithm>
#include <iterator>
#include <string>
#include <unistdx/it/queue_popper>
#include <unistdx/it/queue_pusher>
int main(int arch, char* argv[]) {
std::queue<std::string> q;
std::copy(
sys::cstring_iterator<char*>(argv),
sys::cstring_iterator<char*>(),
sys::queue_pusher(q)
);
std::copy(
sys::queue_popper(q),
sys::queue_popper(),
std::ostream_iterator<std::string>(std::cout, "\n")
);
return 0;
}

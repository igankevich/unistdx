ipacket_and_opacket_guard.cc

Use sys::ipacket_guard and sys::opacket_guard for reading and writing packets to packet streams.

#include <unistdx/net/pstream>
int main() {
uint32_t x = 123, y = 0;
sys::pstream str;
try {
sys::opacket_guard<sys::pstream> g(str);
str.begin_packet();
str << x;
str.end_packet();
} catch (const std::exception& err) {
std::cerr << "write error: " << err.what() << std::endl;
}
str.flush();
if (str.read_packet()) {
try {
sys::ipacket_guard g2(str.rdbuf());
str >> y;
} catch (const std::exception& err) {
std::cerr << "read error: " << err.what() << std::endl;
}
}
return 0;
}

137 int
138 sys::socket::error() const noexcept {
139  int ret = 0;
140  int opt = 0;
141  if (!*this) {
142  ret = -1;
143  } else {
144  try {
145  socklen_type sz = sizeof(opt);
146  UNISTDX_CHECK(
147  ::getsockopt(
148  this->_fd,
149  SOL_SOCKET,
150  SO_ERROR,
151  &opt,
152  &sz
153  )
154  );
155  } catch (...) {
156  ret = -1;
157  }
158  }
159  // ignore EAGAIN since it is common 'error' in asynchronous programming
160  if (opt == EAGAIN || opt == EINPROGRESS) {
161  ret = 0;
162  } else {
163  /*
164  If one connects to localhost to a different port and the service is
165  offline then socket's local port can be chosen to be the same as the
166  port of the service. If this happens the socket connects to itself
167  and sends and replies to its own messages (at least on Linux). This
168  conditional solves the issue.
169  */
170  try {
171  if (ret == 0 && this->name() == this->peer_name()) {
172  ret = -1;
173  }
174  } catch (...) {
175  ret = -1;
176  }
177  }
178  return ret;
179 }

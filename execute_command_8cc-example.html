execute_command.cc

The following code executes errno to display description of error code 42 in Russian locale.

#include <unistdx/ipc/execute>
int main() {
sys::argstream args;
args.append("errno");
args.append(42);
sys::argstream env;
env.append("LANG=ru_RU.UTF-8");
sys::this_process::execute_command(args, env);
return 0;
}
